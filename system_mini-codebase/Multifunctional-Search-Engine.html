        function searchEngine(e) {
            var n = document.querySelector(".search-box").value,
                o = "";
            switch (e) {
                case "bing":
                    o = "https://www.bing.com/search?q=" + encodeURIComponent(n);
                    break;
                case "baidu":
                    o = "https://www.baidu.com/s?wd=" + encodeURIComponent(n);
                    break;
                case "google":
                    o = "https://www.google.com/search?q=" + encodeURIComponent(n);
                    break;
                case "sogou":
                    o = "https://www.sogou.com/web?query=" + encodeURIComponent(n);
                    break;
                case "sm":
                    o = "https://www.sm.cn/s?q=" + encodeURIComponent(n);
                    break;
                case "github":
                    o = "https://github.com/search?q=" + encodeURIComponent(n);
                    break;
                default:
                    return;
            }
            window.open(o, "_blank");
        }
    </script>
